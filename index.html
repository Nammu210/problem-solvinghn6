    var Minimal_Angle = (h, m) => 
    {
      h = h%12;
      let hour = (h*360)/12 + (m*360)/(12*60);
      let min = (m*360)/60;
      let angle = Math.abs(hour-min);
      
      if(angle>180)
      return 360-angle;
      else
      return angle;
    };

      var Perfect_Check = (N) => 
{
      
    let temp =0;
    for (i=1;i<=N/2;i++)
    {
      if(N%i === 0)
      {
        temp += i;
      }
    }
    
    if(temp === N && temp !==0)
    {
      return ("yes");
    }
    else
    {
      return("NO")
    }
    
};
